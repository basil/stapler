001package org.kohsuke.stapler.lang;
002
003import org.kohsuke.stapler.util.IllegalReflectiveAccessLogHandler;
004
005import java.lang.annotation.Annotation;
006import java.lang.reflect.InvocationTargetException;
007import java.lang.reflect.Method;
008import java.lang.reflect.Modifier;
009import java.util.logging.Logger;
010import javax.annotation.CheckForNull;
011
012/**
013 * @author Kohsuke Kawaguchi
014 * @since 1.220
015 */
016public abstract class MethodRef extends AnnotatedRef {
017
018    private static final Logger LOGGER = Logger.getLogger(MethodRef.class.getName());
019
020    /**
021     * Returns true if this method is a 'public' method that should be used for routing requests.
022     */
023    public boolean isRoutable() {
024        return true;
025    }
026    
027    /**
028     * Retrieves the referenced method name.
029     * Some implementations (e.g. Ruby) cannot guarantee availability of names for all cases,
030     * so sometimes the name may be missing.
031     * @return Method name. {@code null} if it cannot be determined.
032     * @since 1.248
033     */
034    @CheckForNull
035    public String getName() {
036        return null;
037    }
038
039    public abstract Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException;
040
041    public static MethodRef wrap(final Method m) {
042        return new MethodRef() {
043            @Override
044            public <T extends Annotation> T getAnnotation(Class<T> type) {
045                return m.getAnnotation(type);
046            }
047
048            @Override
049            public boolean isRoutable() {
050                if (m.isBridge())    return false;
051                return (m.getModifiers() & Modifier.PUBLIC)!=0;
052            }
053      
054            @Override
055            public String getName() {
056                return m.getName();
057            }
058            
059            @Override
060            public Object invoke(Object _this, Object... args) throws InvocationTargetException, IllegalAccessException {
061                try {
062                    return m.invoke(_this, args);
063                } catch (IllegalAccessException e) {
064                    LOGGER.warning(IllegalReflectiveAccessLogHandler.get(e));
065                    m.setAccessible(true);
066                    return m.invoke(_this, args);
067                }
068            }
069        };
070    }
071}
